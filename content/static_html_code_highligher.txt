<script>
const languageSelector = document.getElementById("language-selector")
const editor = document.getElementById("input")
const output = document.getElementById("out")
const preview = document.getElementById("preview")
function parse() {
  const lang = languageSelector[languageSelector.selectedIndex].value;
  const source = editor.value;
  const parsedSource = hljs.highlight(lang, source, false);
  preview.textContent = parsedSource.value;
  <!-- preview.innerHTML = parsedSource.value; -->
}
</script>
